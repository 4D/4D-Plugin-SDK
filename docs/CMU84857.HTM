A log file starts with this structure, then is followed by the number of PA_LogTag described in fNbOperations . 

typedef struct PA_LogHeader
{
   long   fNbOperations;
   long   fLastFlush;
   PA_long64   fLastAction;
   PA_long64   fFirstAction;
   long   fLogNumber;
   long   fFiller;
   long   fLastPos;
} PA_LogHeader;
